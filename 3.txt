using System;

class Program
{

    static int GCD(int a, int b)
    {
        a = Math.Abs(a);
        b = Math.Abs(b);
        while (b != 0)
        {
            int temp = b;
            b = a % b;
            a = temp;
        }
        return a;
    }

    static void Main()
    {
        Console.Write("Введите числитель: ");
        int numerator = Convert.ToInt32(Console.ReadLine());

        int denominator;
        do
        {
            Console.Write("Введите знаменатель (не равен 0): ");
            denominator = Convert.ToInt32(Console.ReadLine());
            if (denominator == 0)
                Console.WriteLine("Знаменатель не может быть равен 0. Попробуйте снова.");
        } while (denominator == 0);

        bool negative = (numerator < 0) ^ (denominator < 0);

        numerator = Math.Abs(numerator);
        denominator = Math.Abs(denominator);

        int gcd = GCD(numerator, denominator);

        numerator /= gcd;
        denominator /= gcd;


        string sign = negative ? "-" : "";
        Console.WriteLine($"Результат: {sign}{numerator} / {denominator}");
    }
}
